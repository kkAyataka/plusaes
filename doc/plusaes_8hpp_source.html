110 inline void shift_rows(State &state) {
111  const State ori = { state[0], state[1], state[2], state[3] };
112  for (int r = 1; r < kWordSize; ++r) {
113  const Word m2 = 0xFF << (r * 8);
114  const Word m1 = ~m2;
115  for (int c = 0; c < kBlockSize; ++c) {
116  state[c] = (state[c] & m1) | (ori[(c + r) % kBlockSize] & m2);
117  }
118  }
119 }
120 
121 inline void inv_shift_rows(State &state) {
122  const State ori = { state[0], state[1], state[2], state[3] };
123  for (int r = 1; r < kWordSize; ++r) {
124  const Word m2 = 0xFF << (r * 8);
125  const Word m1 = ~m2;
126  for (int c = 0; c < kBlockSize; ++c) {
127  state[c] = (state[c] & m1) | (ori[(c + kBlockSize - r) % kWordSize] & m2);
128  }
129  }
130 }

141 inline unsigned char mul(const unsigned char b, const unsigned char m) {
142  unsigned char v = 0;
143  unsigned char t = b;
144  for (int i = 0; i < 8; ++i) { // 8-bits
145  if ((m >> i) & 0x01) {
146  v ^= t;
147  }
148 
149  t = mul2(t);
150  }
151 
152  return v;
153 }

155 inline void mix_columns(State &state) {
156  for (int i = 0; i < kBlockSize; ++i) {
157  const unsigned char v0_1 = (state[i] >> 0) & 0xFF;
158  const unsigned char v1_1 = (state[i] >> 8) & 0xFF;
159  const unsigned char v2_1 = (state[i] >> 16) & 0xFF;
160  const unsigned char v3_1 = (state[i] >> 24) & 0xFF;
161 
162  const unsigned char v0_2 = mul2(v0_1);
163  const unsigned char v1_2 = mul2(v1_1);
164  const unsigned char v2_2 = mul2(v2_1);
165  const unsigned char v3_2 = mul2(v3_1);
166 
167  const unsigned char v0_3 = v0_2 ^ v0_1;
168  const unsigned char v1_3 = v1_2 ^ v1_1;
169  const unsigned char v2_3 = v2_2 ^ v2_1;
170  const unsigned char v3_3 = v3_2 ^ v3_1;
171 
172  state[i] =
173  (v0_2 ^ v1_3 ^ v2_1 ^ v3_1) << 0 |
174  (v0_1 ^ v1_2 ^ v2_3 ^ v3_1) << 8 |
175  (v0_1 ^ v1_1 ^ v2_2 ^ v3_3) << 16 |
176  (v0_3 ^ v1_1 ^ v2_1 ^ v3_2) << 24;
177  }
178 }

180 inline void inv_mix_columns(State &state) {
181  for (int i = 0; i < kBlockSize; ++i) {
182  const unsigned char v0 = (state[i] >> 0) & 0xFF;
183  const unsigned char v1 = (state[i] >> 8) & 0xFF;
184  const unsigned char v2 = (state[i] >> 16) & 0xFF;
185  const unsigned char v3 = (state[i] >> 24) & 0xFF;
186 
187  state[i] =
188  (mul(v0, 0x0E) ^ mul(v1, 0x0B) ^ mul(v2, 0x0D) ^ mul(v3, 0x09)) << 0 |
189  (mul(v0, 0x09) ^ mul(v1, 0x0E) ^ mul(v2, 0x0B) ^ mul(v3, 0x0D)) << 8 |
190  (mul(v0, 0x0D) ^ mul(v1, 0x09) ^ mul(v2, 0x0E) ^ mul(v3, 0x0B)) << 16 |
191  (mul(v0, 0x0B) ^ mul(v1, 0x0D) ^ mul(v2, 0x09) ^ mul(v3, 0x0E)) << 24;
192  }
193 }

220 inline RoundKeys expand_key(const unsigned char *key, const int key_size) {
221  if (key_size != 16 && key_size != 24 && key_size != 32) {
222  throw std::invalid_argument("Invalid key size");
223  }
224 
225  const Word rcon[] = {
226  0x00, 0x01, 0x02, 0x04, 0x08, 0x10,
227  0x20, 0x40, 0x80, 0x1b, 0x36
228  };
229 
230  const int nb = kBlockSize;
231  const int nk = key_size / nb;
232  const int nr = get_round_count(key_size);
233 
234  std::vector<Word> w(nb * (nr + 1));
235  for (int i = 0; i < nk; ++ i) {
236  memcpy(&w[i], key + (i * kWordSize), kWordSize);
237  }
238 
239  for (int i = nk; i < nb * (nr + 1); ++i) {
240  Word t = w[i - 1];
241  if (i % nk == 0) {
242  t = sub_word(rot_word(t)) ^ rcon[i / nk];
243  }
244  else if (nk > 6 && i % nk == 4) {
245  t = sub_word(t);
246  }
247 
248  w[i] = t ^ w[i - nk];
249  }
250 
251  RoundKeys keys(nr + 1);
252  memcpy(&keys[0], &w[0], w.size() * kWordSize);
253 
254  return keys;
255 }

279  unsigned n = kStateSize, c = 1;

288 inline void encrypt_state(const RoundKeys &rkeys, const unsigned char data[16], unsigned char encrypted[16]) {
289  State s;
290  copy_bytes_to_state(data, s);
291 
292  add_round_key(rkeys[0], s);
293 
294  for (unsigned int i = 1; i < rkeys.size() - 1; ++i) {
295  sub_bytes(s);
296  shift_rows(s);
297  mix_columns(s);
298  add_round_key(rkeys[i], s);
299  }
300 
301  sub_bytes(s);
302  shift_rows(s);
303  add_round_key(rkeys.back(), s);
304 
305  copy_state_to_bytes(s, encrypted);
306 }

308 inline void decrypt_state(const RoundKeys &rkeys, const unsigned char data[16], unsigned char decrypted[16]) {
309  State s;
310  copy_bytes_to_state(data, s);
311 
312  add_round_key(rkeys.back(), s);
313  inv_shift_rows(s);
314  inv_sub_bytes(s);
315 
316  for (std::size_t i = rkeys.size() - 2; i > 0; --i) {
317  add_round_key(rkeys[i], s);
318  inv_mix_columns(s);
319  inv_shift_rows(s);
320  inv_sub_bytes(s);
321  }
322 
323  add_round_key(rkeys[0], s);
324 
325  copy_state_to_bytes(s, decrypted);
326 }

383 inline Error check_encrypt_cond(
384  const unsigned long data_size,
385  const unsigned long key_size,
386  const unsigned long encrypted_size,
387  const bool pads) {
388  // check data size
389  if (!pads && (data_size % kStateSize != 0)) {
390  return kErrorInvalidDataSize;
391  }
392 
393  // check key size
394  if (!detail::is_valid_key_size(key_size)) {
395  return kErrorInvalidKeySize;
396  }
397 
398  // check encrypted buffer size
399  if (pads) {
400  const unsigned long required_size = get_padded_encrypted_size(data_size);
401  if (encrypted_size < required_size) {
402  return kErrorInvalidBufferSize;
403  }
404  }
405  else {
406  if (encrypted_size < data_size) {
407  return kErrorInvalidBufferSize;
408  }
409  }
410  return kErrorOk;
411 }

413 inline Error check_decrypt_cond(
414  const unsigned long data_size,
415  const unsigned long key_size,
416  const unsigned long decrypted_size,
417  const unsigned long * padded_size
418  ) {
419  // check data size
420  if (data_size % 16 != 0) {
421  return kErrorInvalidDataSize;
422  }
423 
424  // check key size
425  if (!detail::is_valid_key_size(key_size)) {
426  return kErrorInvalidKeySize;
427  }
428 
429  // check decrypted buffer size
430  if (!padded_size) {
431  if (decrypted_size < data_size) {
432  return kErrorInvalidBufferSize;
433  }
434  }
435  else {
436  if (decrypted_size < (data_size - kStateSize)) {
437  return kErrorInvalidBufferSize;
438  }
439  }
440 
441  return kErrorOk;
442 }

444 inline bool check_padding(const unsigned long padding, const unsigned char data[kStateSize]) {
445  if (padding > kStateSize) {
446  return false;
447  }
448 
449  for (unsigned long i = 0; i < padding; ++i) {
450  if (data[kStateSize - 1 - i] != padding) {
451  return false;
452  }
453  }
454 
455  return true;
456 }

475 inline Error encrypt_ecb(
476  const unsigned char * data,
477  const unsigned long data_size,
478  const unsigned char * key,
479  const unsigned long key_size,
480  unsigned char *encrypted,
481  const unsigned long encrypted_size,
482  const bool pads
483  ) {
484  const Error e = detail::check_encrypt_cond(data_size, key_size, encrypted_size, pads);
485  if (e != kErrorOk) {
486  return e;
487  }
488 
489  const detail::RoundKeys rkeys = detail::expand_key(key, static_cast<int>(key_size));
490 
491  const unsigned long bc = data_size / detail::kStateSize;
492  for (unsigned long i = 0; i < bc; ++i) {
493  detail::encrypt_state(rkeys, data + (i * detail::kStateSize), encrypted + (i * detail::kStateSize));
494  }
495 
496  if (pads) {
497  const int rem = data_size % detail::kStateSize;
498  const char pad_v = detail::kStateSize - rem;
499 
500  std::vector<unsigned char> ib(detail::kStateSize, pad_v), ob(detail::kStateSize);
501  memcpy(&ib[0], data + data_size - rem, rem);
502 
503  detail::encrypt_state(rkeys, &ib[0], &ob[0]);
504  memcpy(encrypted + (data_size - rem), &ob[0], detail::kStateSize);
505  }
506 
507  return kErrorOk;
508 }

523 inline Error decrypt_ecb(
524  const unsigned char * data,
525  const unsigned long data_size,
526  const unsigned char * key,
527  const unsigned long key_size,
528  unsigned char * decrypted,
529  const unsigned long decrypted_size,
530  unsigned long * padded_size
531  ) {
532  const Error e = detail::check_decrypt_cond(data_size, key_size, decrypted_size, padded_size);
533  if (e != kErrorOk) {
534  return e;
535  }
536 
537  const detail::RoundKeys rkeys = detail::expand_key(key, static_cast<int>(key_size));
538 
539  const unsigned long bc = data_size / detail::kStateSize - 1;
540  for (unsigned long i = 0; i < bc; ++i) {
541  detail::decrypt_state(rkeys, data + (i * detail::kStateSize), decrypted + (i * detail::kStateSize));
542  }
543 
544  unsigned char last[detail::kStateSize] = {};
545  detail::decrypt_state(rkeys, data + (bc * detail::kStateSize), last);
546 
547  if (padded_size) {
548  *padded_size = last[detail::kStateSize - 1];
549  const unsigned long cs = detail::kStateSize - *padded_size;
550 
551  if (!detail::check_padding(*padded_size, last)) {
552  return kErrorInvalidKey;
553  }
554  else if (decrypted_size >= (bc * detail::kStateSize) + cs) {
555  memcpy(decrypted + (bc * detail::kStateSize), last, cs);
556  }
557  else {
558  return kErrorInvalidBufferSize;
559  }
560  }
561  else {
562  memcpy(decrypted + (bc * detail::kStateSize), last, sizeof(last));
563  }
564 
565  return kErrorOk;
566 }
